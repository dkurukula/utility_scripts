Below assumes the db name is database.accdb
#+BEGIN_SRC bash
mdb-tables -d ',' database.accdb| xargs -L1 -d',' -I{} bash -c 'mdb-export database.accdb "$1" >"$1".csv' -- {}
#+END_SRC
